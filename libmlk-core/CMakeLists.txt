	NAME libmlk-core
	ASSETS ${ASSETS}
	SOURCES ${SOURCES}
	HEADERS ${HEADERS}
	HEADERS_DIRECTORY mlk/core
	LANGS ${NLS}
	INSTALL
	LIBRARIES
		PUBLIC
			${LIBRARIES}
			${OPENAL_LIBRARY}
			SndFile::SndFile
			SDL3::SDL3
			SDL3_image::SDL3_image-shared
			SDL3_ttf::SDL3_ttf-shared
			libmlk-util
	INCLUDES
		PRIVATE
			$<BUILD_INTERFACE:${libmlk-core_BINARY_DIR}>
		PUBLIC
			$<BUILD_INTERFACE:${libmlk-core_SOURCE_DIR}>
			${OPENAL_INCLUDE_DIR}
)
